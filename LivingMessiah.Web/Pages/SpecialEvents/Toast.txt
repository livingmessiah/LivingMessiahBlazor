
using Blazored.Toast.Services;

[Inject] public IToastService Toast { get; set; }


	private async Task Populate() // int id
	{
		try
		{
			DataSource = await db.GetDataSource();
			if (DataSource == null)
			{
				string s = $"DataSource is null after calling {nameof(db.GetDataSource)}";
				Logger.LogWarning(string.Format("...{0}, Sql:{1}", s, db.BaseSqlDump));
				Toast.ShowWarning($"...{s}");
			}
			else
			{
				Toast.ShowInfo("Got Data Source");
			}

		}
		catch (Exception ex)
		{
			Logger.LogError(ex, string.Format("...Inside catch of {0}"
				, nameof(ClassName) + "!" + nameof(Populate)));
			Toast.ShowError("An invalid operation occurred reading database, contact your administrator");
		}
  }
